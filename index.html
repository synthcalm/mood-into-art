const express = require('express'); const cors = require('cors'); const multer = require('multer'); const { SpeechClient } = require('@google-cloud/speech'); const { TextToImageClient } = require('@stabilityai/stable-diffusion'); const fs = require('fs'); const path = require('path'); const app = express(); const upload = multer({ storage: multer.memoryStorage() }); // Middleware app.use(cors()); // Enable CORS for all origins app.use(express.json()); // Google Speech-to-Text Client const speechClient = new SpeechClient({ keyFilename: process.env.GOOGLE_APPLICATION_CREDENTIALS || '/etc/secrets/GOOGLE_APPLICATION_CREDENTIALS' }); // Stability AI Client const stabilityClient = new TextToImageClient({ apiKey: process.env.STABILITY_API_KEY }); // Health Check Endpoint app.get('/health', (req, res) => { res.status(200).json({ status: 'OK' }); }); // Transcribe Endpoint app.post('/transcribe', upload.single('audio'), async (req, res) => { try { if (!req.file) { return res.status(400).json({ error: 'No audio file provided' }); } const audioBytes = req.file.buffer.toString('base64'); const audio = { content: audioBytes }; const config = { encoding: 'WEBM_OPUS', sampleRateHertz: 48000, languageCode: 'en-US' }; const request = { audio: audio, config: config }; const [response] = await speechClient.recognize(request); const transcription = response.results .map(result => result.alternatives[0].transcript) .join('\n'); res.status(200).json({ transcription }); } catch (error) { console.error('Transcription error:', error); res.status(500).json({ error: 'Failed to transcribe audio: ' + error.message }); } }); // Generate Image Endpoint app.post('/generate', async (req, res) => { try { const { prompt } = req.body; if (!prompt) { return res.status(400).json({ error: 'Prompt is required' }); } const image = await stabilityClient.generate({ prompt, width: 512, height: 512 }); const imageBase64 = image.toBase64(); res.status(200).json({ image: `data:image/png;base64,${imageBase64}` }); } catch (error) { console.error('Image generation error:', error); res.status(500).json({ error: 'Failed to generate image: ' + error.message }); } }); // Start Server const PORT = process.env.PORT || 10000; app.listen(PORT, () => { console.log(`Server listening on port ${PORT}`); });
